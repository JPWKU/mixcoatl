[docs]    def rules(self):
        try:
            return self.__rules
        except AttributeError:
            from mixcoatl.network.firewall_rule import FirewallRule
            rls = FirewallRule.all(self.__firewall_id)
            if len(rls) < 1:
                self.__rules = []
            else:
                self.__rules = rls
            return rls
