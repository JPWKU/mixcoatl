[docs]    def all(cls, region_id=None):
        r = Resource(cls.path)
        if region_id is not None:
            params = {'regionId':region_id}
            c = r.get(params=params)
        else:
            c = r.get()
        if r.last_error is None:
            return [cls(i['snapshotId']) for i in c[cls.collection_name]]
        else:
            return r.last_error
